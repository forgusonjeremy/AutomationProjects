$VcfaFqdn = "your-vcfa-fqdn"
$Username  = "admin"
$Password  = "your-password"

# Skip cert validation
add-type @"
    using System.Net;
    using System.Security.Cryptography.X509Certificates;
    public class TrustAllCerts : ICertificatePolicy {
        public bool CheckValidationResult(ServicePoint sp, X509Certificate cert, WebRequest req, int prob) { return true; }
    }
"@
[System.Net.ServicePointManager]::CertificatePolicy = New-Object TrustAllCerts
[System.Net.ServicePointManager]::SecurityProtocol = [Net.SecurityProtocolType]::Tls12

# Auth - VCF Automation 9 uses Basic auth to get a bearer token
$Bytes  = [System.Text.Encoding]::UTF8.GetBytes("${Username}@system:${Password}")
$Base64 = [Convert]::ToBase64String($Bytes)
$Headers = @{
    "Authorization" = "Basic $Base64"
    "Accept"        = "application/json;version=9.0.0"
    "Content-Type"  = "application/json;version=9.0.0"
}

# Get session token
$Session = Invoke-RestMethod -Uri "https://$VcfaFqdn/cloudapi/1.0.0/sessions/provider" `
    -Method POST -Headers $Headers
$Token = $Session.access_token

$AuthHeaders = @{
    "Authorization" = "Bearer $Token"
    "Accept"        = "application/json;version=9.0.0"
}

# Check Supervisors
Write-Host "`n--- Supervisors ---" -ForegroundColor Cyan
Invoke-RestMethod -Uri "https://$VcfaFqdn/cloudapi/1.0.0/supervisors" `
    -Method GET -Headers $AuthHeaders | ConvertTo-Json -Depth 5

# Check vCenters
Write-Host "`n--- vCenters ---" -ForegroundColor Cyan
Invoke-RestMethod -Uri "https://$VcfaFqdn/cloudapi/1.0.0/vcenters" `
    -Method GET -Headers $AuthHeaders | ConvertTo-Json -Depth 5

# Check Regions
Write-Host "`n--- Regions ---" -ForegroundColor Cyan
Invoke-RestMethod -Uri "https://$VcfaFqdn/cloudapi/1.0.0/regions" `
    -Method GET -Headers $AuthHeaders | ConvertTo-Json -Depth 5
