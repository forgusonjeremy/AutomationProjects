// Bind vcenterSdkConnection (VC:SdkConnection)

var cluster = vcenterSdkConnection.rootFolder.childEntity[0]
                .hostFolder.childEntity[0];
var vis     = cluster.host[0].configManager.vsanInternalSystem;

// ── DOM owner stats -- explore the flat structure ─────────────────────────
System.log("=== dom.owner.stats structure ===");
var domRaw    = vis.queryVsanStatistics(["dom"]);
var domParsed = JSON.parse(domRaw);
var ownerStats = domParsed["dom.owner.stats"];

// It's a flat object -- log all its keys and value types
var keys = Object.keys(ownerStats);
System.log("Total keys: " + keys.length);
System.log("First 30 keys and values:");
for (var i = 0; i < Math.min(30, keys.length); i++) {
    var k = keys[i];
    var v = ownerStats[k];
    System.log("  [" + i + "] " + k + " = " + 
        (typeof v === "object" ? JSON.stringify(v).substring(0, 100) : v) +
        "  (type=" + typeof v + ")");
}

// ── LSOM disks -- confirm the latency/outstanding fields ─────────────────
System.log("\n=== lsom.disks latency fields ===");
var lsomRaw    = vis.queryVsanStatistics(["lsom"]);
var lsomParsed = JSON.parse(lsomRaw);
var disks      = lsomParsed["lsom.disks"];
var diskIds    = Object.keys(disks);

var totalReadOutstanding  = 0;
var totalWriteOutstanding = 0;
var totalReadLatMean      = 0;
var totalWriteLatMean     = 0;
var diskCount             = 0;

for (var di = 0; di < diskIds.length; di++) {
    var dk = diskIds[di];
    var d  = disks[dk];

    var rOut  = d.rcReadQStats  && d.rcReadQStats.qStats  ? d.rcReadQStats.qStats.totalOutstanding   : 0;
    var wOut  = d.rcWriteQStats && d.rcWriteQStats.qStats ? d.rcWriteQStats.qStats.totalOutstanding  : 0;
    var rLat  = d.rcReadQStats  && d.rcReadQStats.rcIOStats && d.rcReadQStats.rcIOStats.latencyHisto
                ? d.rcReadQStats.rcIOStats.latencyHisto.mean : 0;
    var wLat  = d.rcWriteQStats && d.rcWriteQStats.rcIOStats && d.rcWriteQStats.rcIOStats.latencyHisto
                ? d.rcWriteQStats.rcIOStats.latencyHisto.mean : 0;

    System.log("Disk " + dk + ":");
    System.log("  rcReadQStats.qStats.totalOutstanding  = " + rOut);
    System.log("  rcWriteQStats.qStats.totalOutstanding = " + wOut);
    System.log("  rcReadQStats.rcIOStats.latencyHisto.mean  = " + rLat + " us");
    System.log("  rcWriteQStats.rcIOStats.latencyHisto.mean = " + wLat + " us");

    // Also check stats sub-object if present
    if (d.stats) {
        System.log("  stats keys: " + Object.keys(d.stats).join(", "));
        System.log("  stats sample: " + JSON.stringify(d.stats).substring(0, 300));
    }

    totalReadOutstanding  += rOut;
    totalWriteOutstanding += wOut;
    totalReadLatMean      += rLat;
    totalWriteLatMean     += wLat;
    diskCount++;
}

System.log("\n=== Aggregated across " + diskCount + " disks ===");
System.log("Total read outstanding:  " + totalReadOutstanding);
System.log("Total write outstanding: " + totalWriteOutstanding);
System.log("Avg read latency (mean of disk means):  " + 
    (diskCount > 0 ? (totalReadLatMean  / diskCount / 1000).toFixed(3) : "n/a") + " ms");
System.log("Avg write latency (mean of disk means): " + 
    (diskCount > 0 ? (totalWriteLatMean / diskCount / 1000).toFixed(3) : "n/a") + " ms");

// ── Also check dom.client.stats for VM-facing latency ────────────────────
System.log("\n=== dom.client.stats ===");
var clientStats = domParsed["dom.client.stats"];
if (clientStats && typeof clientStats === "object") {
    var cKeys = Object.keys(clientStats);
    System.log("Total keys: " + cKeys.length);
    System.log("First 20 keys:");
    for (var ci = 0; ci < Math.min(20, cKeys.length); ci++) {
        var ck = cKeys[ci];
        var cv = clientStats[ck];
        System.log("  " + ck + " = " +
            (typeof cv === "object" ? JSON.stringify(cv).substring(0, 150) : cv) +
            "  (type=" + typeof cv + ")");
    }
}
