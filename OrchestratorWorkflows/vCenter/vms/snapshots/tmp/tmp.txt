# Put the anti-auto-default NM config in place (persists across boots — this is RPM-safe)
sudo tee /etc/NetworkManager/conf.d/10-no-auto-default.conf <<'EOF'
[main]
no-auto-default=*
EOF

# Put the VMware datasource config in place
sudo tee /etc/cloud/cloud.cfg.d/99-vmware-guestinfo.cfg <<'EOF'
datasource_list: [ VMware, OVF, None ]
datasource:
  VMware:
    allow_raw_data: true
EOF

# Now the cleanup — do this LAST, immediately before shutdown
sudo rm -f /etc/NetworkManager/system-connections/*.nmconnection
sudo cloud-init clean --logs --seed
sudo rm -rf /var/lib/cloud/*
sudo rm -f /etc/machine-id /var/lib/dbus/machine-id
sudo touch /etc/machine-id
sudo hostnamectl set-hostname localhost.localdomain

# Shutdown immediately
sudo shutdown -h now
