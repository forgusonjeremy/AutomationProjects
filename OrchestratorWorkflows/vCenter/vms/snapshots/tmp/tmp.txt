// Bind vcenterSdkConnection (VC:SdkConnection)

var cluster = vcenterSdkConnection.rootFolder.childEntity[0]
                .hostFolder.childEntity[0];
var host    = cluster.host[0];
var vis     = host.configManager.vsanInternalSystem;

// The previous run returned dom.owners.stats with real data.
// That must have come from a different label. Try all known vSAN stat labels.
var candidates = [
    "dom-owners",
    "dom-clients", 
    "dom",
    "lsom",
    "lsom-world-md",
    "vsan-disk-stats",
    "vsan-pnic-stats",
    "rdtAssoc",
    "rdtTransport",
    "plog",
    "plog-lsom-worlds",
    "dom-proxy-owners",
    "clom-disk-stats",
    "clom-host-stats",
    "sos",
    "mem-stats",
    "vsan-mgmt-lwd",
    "version",
    "system-mem-stats"
];

for (var li = 0; li < candidates.length; li++) {
    var label = candidates[li];
    try {
        var result = vis.queryVsanStatistics([label]);
        var parsed = JSON.parse(result);
        var keys   = Object.keys(parsed);
        // Filter out boring single-key results that are just timing info
        var interesting = keys.filter(function(k) {
            return k !== "on-esx-collect-duration";
        });
        if (interesting.length > 0) {
            System.log("LABEL '" + label + "' -> keys: " + interesting.join(", "));
            // If it has dom stats, show a sample
            for (var ki = 0; ki < interesting.length; ki++) {
                var val = parsed[interesting[ki]];
                if (typeof val === "object" && val !== null) {
                    var subKeys = Object.keys(val);
                    System.log("  " + interesting[ki] + " -> " +
                        subKeys.length + " entries, sample keys: " +
                        (subKeys.length > 0
                            ? Object.keys(val[subKeys[0]]).slice(0,8).join(", ")
                            : "(empty)"));
                } else {
                    System.log("  " + interesting[ki] + " = " + val);
                }
            }
        } else {
            System.log("LABEL '" + label + "' -> no interesting data");
        }
    } catch (e) {
        System.log("LABEL '" + label + "' -> ERROR: " + e.message);
    }
}
