var vc = // bind your vcenterSdkConnection here
var perfMgr = vc.performanceManager;
var found = [];
for (var i = 0; i < perfMgr.perfCounter.length; i++) {
    var c = perfMgr.perfCounter[i];
    if (c.groupInfo.key.indexOf("vsanDomObj") !== -1 ||
        c.groupInfo.key.indexOf("vsan") !== -1) {
        found.push(c.groupInfo.key + ":" + c.nameInfo.key + ":" + c.rollupType.value);
    }
}
System.log("vSAN counters found: " + found.sort().join("\n"));
