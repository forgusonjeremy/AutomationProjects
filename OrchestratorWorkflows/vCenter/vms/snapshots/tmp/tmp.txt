var taskState = "";
var maxWait = 300000; // 5 minute timeout per snapshot task
var waited = 0;
var pollInterval = 5000; // check every 5 seconds

while (taskState !== "success" && taskState !== "error" && waited < maxWait) {
    System.sleep(pollInterval);
    waited += pollInterval;
    taskState = task.info.state.value;
}

if (taskState === "error") {
    throw new Error("Snapshot task failed: " + task.info.error.localizedMessage);
} else if (waited >= maxWait) {
    throw new Error("Snapshot task timed out after " + (maxWait/60000) + " minutes");
}
