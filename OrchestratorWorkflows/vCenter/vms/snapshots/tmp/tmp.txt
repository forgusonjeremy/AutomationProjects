// Bind vcenterSdkConnection (VC:SdkConnection)

var cluster = vcenterSdkConnection.rootFolder.childEntity[0]
                .hostFolder.childEntity[0];

// Query dom label across all hosts and aggregate
System.log("=== DOM stats across all hosts ===");

for (var hi = 0; hi < cluster.host.length; hi++) {
    var host = cluster.host[hi];
    var vis  = host.configManager.vsanInternalSystem;

    try {
        var raw    = vis.queryVsanStatistics(["dom"]);
        var parsed = JSON.parse(raw);

        System.log("\nHost: " + host.name);

        // dom.owner.stats -- aggregate owner-level latency/congestion
        var ownerStats = parsed["dom.owner.stats"];
        if (ownerStats && typeof ownerStats === "object") {
            var uuids = Object.keys(ownerStats);
            System.log("  dom.owner.stats: " + uuids.length + " objects");
            if (uuids.length > 0) {
                var sample = ownerStats[uuids[0]];
                System.log("  Sample object UUID: " + uuids[0]);
                System.log("  Sample fields: " + Object.keys(sample).join(", "));
                System.log("  Sample values: " + JSON.stringify(sample).substring(0, 600));
            }
        } else {
            System.log("  dom.owner.stats type: " + typeof ownerStats);
        }

        // dom.client.stats -- client-side latency seen by VMs
        var clientStats = parsed["dom.client.stats"];
        if (clientStats && typeof clientStats === "object") {
            var cUuids = Object.keys(clientStats);
            System.log("  dom.client.stats: " + cUuids.length + " objects");
            if (cUuids.length > 0) {
                var cSample = clientStats[cUuids[0]];
                System.log("  Client sample fields: " + Object.keys(cSample).join(", "));
                System.log("  Client sample values: " + JSON.stringify(cSample).substring(0, 600));
            }
        }

    } catch (e) {
        System.log("Host " + host.name + " error: " + e.message);
    }
}

// Also drill into lsom.disks for latency stats
System.log("\n=== LSOM disk stats (first host) ===");
try {
    var host0  = cluster.host[0];
    var vis0   = host0.configManager.vsanInternalSystem;
    var lraw   = vis0.queryVsanStatistics(["lsom"]);
    var lp     = JSON.parse(lraw);
    var disks  = lp["lsom.disks"];
    if (disks) {
        var diskIds = Object.keys(disks);
        System.log("Disk count: " + diskIds.length);
        if (diskIds.length > 0) {
            var d = disks[diskIds[0]];
            System.log("Disk id: " + diskIds[0]);
            System.log("Disk top-level keys: " + Object.keys(d).join(", "));
            // stats sub-object
            if (d.stats) {
                System.log("d.stats keys: " + Object.keys(d.stats).join(", "));
                System.log("d.stats values: " + JSON.stringify(d.stats).substring(0, 600));
            }
            // rcReadQStats
            if (d.rcReadQStats) {
                System.log("rcReadQStats: " + JSON.stringify(d.rcReadQStats));
            }
            if (d.rcWriteQStats) {
                System.log("rcWriteQStats: " + JSON.stringify(d.rcWriteQStats));
            }
        }
    }
} catch (e2) {
    System.log("lsom error: " + e2.message);
}
