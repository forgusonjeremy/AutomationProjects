// Run this as a standalone scriptable task
// Bind vcenterSdkConnection as an input (VC:SdkConnection)

var perfMgr = vcenterSdkConnection.performanceManager;
var allCounters = perfMgr.perfCounter;

// Find vsanDomObj counters and log them
var vsanCounters = [];
for (var i = 0; i < allCounters.length; i++) {
    var c = allCounters[i];
    if (c.groupInfo.key === "vsanDomObj") {
        vsanCounters.push({
            key:    c.key,
            name:   c.groupInfo.key + ":" + c.nameInfo.key + ":" + c.rollupType.value,
            level:  c.level,
            perDev: c.perDeviceLevel
        });
    }
}
System.log("vsanDomObj counter count: " + vsanCounters.length);

// Now find what entity types actually have perf data available
// by checking the performance provider summary for each entity type
var providerSummary = perfMgr.queryPerfProviderSummary(
    vcenterSdkConnection.rootFolder
);
System.log("rootFolder provider summary: " +
    "currentSupported=" + providerSummary.currentSupported +
    " summarySupported=" + providerSummary.summarySupported +
    " refreshRate=" + providerSummary.refreshRate);

// Check a host entity -- hosts are the natural home for vsanDomObj counters
var allHosts = vcenterSdkConnection.rootFolder.childEntity[0].hostFolder.childEntity;
System.log("Host count in first datacenter: " + allHosts.length);

if (allHosts.length > 0) {
    var firstCluster = allHosts[0];
    System.log("First cluster/host type: " + firstCluster.vimType);
    
    // If it's a cluster, get a host from it
    var testEntity = firstCluster;
    if (firstCluster.host) {
        testEntity = firstCluster.host[0];
        System.log("Testing with host: " + testEntity.name);
    }
    
    // Check if this entity supports vsanDomObj counters
    var hostSummary = perfMgr.queryPerfProviderSummary(testEntity);
    System.log("Host provider summary: " +
        "currentSupported=" + hostSummary.currentSupported +
        " summarySupported=" + hostSummary.summarySupported +
        " refreshRate=" + hostSummary.refreshRate);
    
    // Try to actually query vsanDomObj against a host
    if (vsanCounters.length > 0 && hostSummary.currentSupported) {
        var mid = new VcPerfMetricId();
        mid.counterId = vsanCounters[0].key;
        mid.instance  = "*";  // wildcard to get all instances
        
        var now   = new Date();
        var start = new Date(now.getTime() - 300000);
        
        var qs        = new VcPerfQuerySpec();
        qs.entity     = testEntity;
        qs.startTime  = start;
        qs.endTime    = now;
        qs.intervalId = 300;
        qs.metricId   = [mid];
        qs.maxSample  = 1;
        
        try {
            var results = perfMgr.queryPerf([qs]);
            System.log("queryPerf result count against host: " +
                (results ? results.length : 0));
            if (results && results.length > 0 && results[0].value) {
                System.log("Data returned: " + results[0].value.length + " series");
                for (var ri = 0; ri < Math.min(3, results[0].value.length); ri++) {
                    var s = results[0].value[ri];
                    System.log("  instance='" + s.id.instance +
                               "' counterId=" + s.id.counterId +
                               " values=" + (s.value ? s.value.length : 0));
                }
            } else {
                System.log("No data returned for host entity");
            }
        } catch (e) {
            System.log("queryPerf against host failed: " + e.message);
        }
    }
}
