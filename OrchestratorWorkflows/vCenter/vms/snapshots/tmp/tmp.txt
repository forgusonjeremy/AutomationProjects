var cluster = vcenterSdkConnection.rootFolder.childEntity[0]
                .hostFolder.childEntity[0];

System.log("Cluster: " + cluster.name);

var hosts = cluster.host;
var host  = hosts[0];
System.log("Host: " + host.name);

var cfg = host.configManager;

// Access specific configManager properties directly -- no Object.keys()
System.log("vsanInternalSystem: " + (cfg.vsanInternalSystem ? "present" : "null"));
System.log("vsanSystem: "         + (cfg.vsanSystem         ? "present" : "null"));

if (cfg.vsanInternalSystem) {
    var vis = cfg.vsanInternalSystem;
    System.log("vsanInternalSystem.vimType: " + vis.vimType);
    
    // Try queryCmmds -- returns raw vSAN stats including performance data
    try {
        var queries = '[{"type":"DOM_OBJECT","fields":["read_congestion","write_congestion","read_count","write_count"]}]';
        var result  = vis.queryCmmds(queries);
        System.log("queryCmmds result length: " + (result ? result.length : "null"));
        System.log("queryCmmds sample: " + String(result).substring(0, 500));
    } catch (e1) {
        System.log("queryCmmds failed: " + e1.message);
    }
    
    // Try queryVsanStatistics
    try {
        var labels = ["dom-objects"];
        var stats  = vis.queryVsanStatistics(labels);
        System.log("queryVsanStatistics result: " + String(stats).substring(0, 500));
    } catch (e2) {
        System.log("queryVsanStatistics failed: " + e2.message);
    }
}

if (cfg.vsanSystem) {
    var vs = cfg.vsanSystem;
    System.log("vsanSystem.vimType: " + vs.vimType);
    System.log("vsanSystem.config: " + JSON.stringify(vs.config));
}

// serviceContent -- access specific known vSAN properties
System.log("\n--- serviceContent vSAN properties ---");
var sc = vcenterSdkConnection.serviceContent;
System.log("vsanPerformanceManager: " +
    (sc.vsanPerformanceManager ? "present" : "null"));
System.log("vsanClusterHealthSystem: " +
    (sc.vsanClusterHealthSystem ? "present" : "null"));

// queryAvailablePerfMetric on vSAN datastore
System.log("\n--- queryAvailablePerfMetric on vSAN datastore ---");
var perfMgr    = vcenterSdkConnection.performanceManager;
var datastores = vcenterSdkConnection.getAllDatastores();
for (var di = 0; di < datastores.length; di++) {
    var ds = datastores[di];
    if (ds.summary && ds.summary.type &&
        ds.summary.type.toLowerCase() === "vsan") {
        System.log("vSAN datastore: " + ds.name + " id=" + ds.id);
        
        try {
            var avail20 = perfMgr.queryAvailablePerfMetric(ds, null, null, 20);
            System.log("availableMetrics intervalId=20: " +
                (avail20 ? avail20.length : 0));
            if (avail20 && avail20.length > 0) {
                for (var ai = 0; ai < Math.min(5, avail20.length); ai++) {
                    System.log("  counterId=" + avail20[ai].counterId +
                               " instance='" + avail20[ai].instance + "'");
                }
            }
        } catch (e3) {
            System.log("queryAvailablePerfMetric(20) error: " + e3.message);
        }

        try {
            var avail300 = perfMgr.queryAvailablePerfMetric(ds, null, null, 300);
            System.log("availableMetrics intervalId=300: " +
                (avail300 ? avail300.length : 0));
            if (avail300 && avail300.length > 0) {
                for (var bi = 0; bi < Math.min(5, avail300.length); bi++) {
                    System.log("  counterId=" + avail300[bi].counterId +
                               " instance='" + avail300[bi].instance + "'");
                }
            }
        } catch (e4) {
            System.log("queryAvailablePerfMetric(300) error: " + e4.message);
        }
        break;
    }
}
