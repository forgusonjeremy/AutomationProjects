# Update datasource config - OVF first
sudo tee /etc/cloud/cloud.cfg.d/99-vmware-guestinfo.cfg <<'EOF'
datasource_list: [ OVF, VMware, None ]
datasource:
  OVF:
    allow_raw_data: true
  VMware:
    allow_raw_data: true
EOF

# Verify it's saved
sudo cat /etc/cloud/cloud.cfg.d/99-vmware-guestinfo.cfg

# Clean state
sudo rm -f /etc/NetworkManager/system-connections/*.nmconnection
sudo cloud-init clean --logs --seed
sudo rm -rf /var/lib/cloud/*
sudo rm -rf /var/log/vmware-imc/*
sudo rm -f /etc/machine-id /var/lib/dbus/machine-id
sudo touch /etc/machine-id
sudo hostnamectl set-hostname localhost.localdomain
history -c
sudo shutdown -h now
