# Disable VMware customization in cloud-init
sudo sed -i 's/^disable_vmware_customization:.*/disable_vmware_customization: true/' /etc/cloud/cloud.cfg
# If the line doesn't exist, append it:
grep -q "disable_vmware_customization" /etc/cloud/cloud.cfg || echo "disable_vmware_customization: true" | sudo tee -a /etc/cloud/cloud.cfg

# Datasource config — keep OVF first since that's what works for user-data in your VCF version
sudo tee /etc/cloud/cloud.cfg.d/99-vmware-guestinfo.cfg <<'EOF'
datasource_list: [ OVF, VMware, None ]
datasource:
  OVF:
    allow_raw_data: true
  VMware:
    allow_raw_data: true
EOF

# Standard cleanup
sudo rm -f /etc/NetworkManager/system-connections/*.nmconnection
sudo cloud-init clean --logs --seed
sudo rm -rf /var/lib/cloud/*
sudo rm -rf /var/log/vmware-imc/*
sudo rm -f /etc/machine-id /var/lib/dbus/machine-id
sudo touch /etc/machine-id
sudo hostnamectl set-hostname localhost.localdomain
history -c
sudo shutdown -h now
